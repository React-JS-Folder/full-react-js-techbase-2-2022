1. Create a react app  // npx create-react-app  <<name>>
2. Install react query // npm i react-query 


3. Create a query client and wrap anything you want 

index.js
            import {QueryClient, QueryClientProvider} from 'react-query'

            const queryClient = new QueryClient();


            ReactDOM.render(
   
               <QueryClientProvider client={queryClient} >
                  <App />
               </QueryClientProvider>
            )

-------------------------------------------------
